# KannadaDotCom MongoDB Schema

Collection: users

Fields:

- name (String)
- email (String)
- password (String)

Collection: posts

Fields:

- id (String)

- business
  - en (String)
  - kn (String)

- location
  - en (String)
  - kn (String)

- content
  - en (String)
  - kn (String)

- tags (Array)
  - en (String)
  - kn (String)

- likes (Number)
- comments (Number)
- saves (Number)

- image (String)
- createdAt (Number)
